SQL Export Query

SELECT 
    j.name as job_name,
    cc.company_name,
    cc.domain,
    cc.hubspot_object_id,
    pc.query_name as persona,
    pc.total_count,
    he.vertical
FROM jobs j
JOIN csv_companies cc ON cc.job_id = j.id
LEFT JOIN person_counts pc ON pc.csv_company_id = cc.id AND pc.is_active = true
LEFT JOIN hubspot_enrichments he ON he.csv_company_id = cc.id AND he.is_active = true
WHERE j.id = {{ job.id }}
ORDER BY cc.company_name, pc.query_name;
